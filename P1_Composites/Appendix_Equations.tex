\[
\rho_c = f\,\rho_f + (1-f)\,\rho_m
\]

\[
f = \frac{\rho_c - \rho_m}{\rho_f - \rho_m}
\]

\[
\delta = \frac{F s \left(3L^2 - 4s^2\right)}{48\,E\,I}
\]

\[
I = \frac{w h^3}{12}
\]

\[
E_f^{\mathrm{ax}} = \text{Axial fibre modulus}
\]

\[
E_1 = \text{Axial composite modulus}
\]

\[
E_1 = f E_f^{ax} + (1-f) E_m
\]

\[
E_f^{ax} = \frac{E_1 - (1-f)E_m}{f}
\]

\[
E_f^{\mathrm{tr}} = \text{Transverse fibre modulus}
\]

\[
E_2 = \text{Transverse composite modulus}
\]

\[
E_2 = E_m\,\frac{1+\eta f}{1-\eta f}
\]

\[
\eta = \frac{\left(\dfrac{E_f^{\mathrm{tr}}}{E_m}\right) - 1}
           {\left(\dfrac{E_f^{\mathrm{tr}}}{E_m}\right) + 1}
     = \frac{E_f^{\mathrm{tr}} - E_m}{E_f^{\mathrm{tr}} + E_m}
\]

\[
E_f^{\mathrm{tr}}
= E_m \frac{1+\eta}{1-\eta}
\]
